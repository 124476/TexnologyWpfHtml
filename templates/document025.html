ChartValues< double > doubles = new ChartValues< double >();
foreach (var item in App.DB.Grafik)
{
    doubles.Add(item.Num);
}

var seriaes = new LiveCharts.SeriesCollection()
{
    new LineSeries
    {
        Title = "Grafik",
        Values = doubles,
        Fill = null
    }
};

Cartes.Series = seriaes;
